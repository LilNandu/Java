
import java.lang.*;

public class MathFunctions { 
    public static void main(String[] args) { 
        double x = 28; 
        double y = 4;

        System.out.println("Maximum number of x and y is: " + Math.max(x, y)); 
        System.out.println("Square root of y is: " + Math.sqrt(y)); 
        System.out.println("Power of x raised to y is: " + Math.pow(x, y)); 
        System.out.println("Logarithm value of x is: " + Math.log(x)); 
        System.out.println("Log10 value of x is: " + Math.log10(x)); 
        System.out.println("Exponential of x is: " + Math.exp(x)); 
    }
}
